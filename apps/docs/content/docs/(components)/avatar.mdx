### Usage

```tsx
import { Avatar, AvatarFallback, AvatarImage } from '~/components/ui/avatar';
import { Text } from '~/components/ui/text';

```

```tsx
<Avatar alt="Zach Nugent's Avatar">
  <AvatarImage source={{ uri: 'https://github.com/mrzachnugent.png' }} />
  <AvatarFallback>
    <Text>ZN</Text>
  </AvatarFallback>
</Avatar>
```
